\begin{frame}{Operacoes Principais da Stack}
  \begin{columns}[t]
    \column{0.31\textwidth}
      \begin{exampleblock}{push(x)}
        \small
        Insere \texttt{x} no topo.\\[4pt]
        \textbf{O(1)} sempre.\\[4pt]
        Nunca falha por si so.
      \end{exampleblock}
    \column{0.31\textwidth}
      \begin{alertblock}{pop()}
        \small
        Remove o topo.\\[4pt]
        \textbf{O(1)}\\[4pt]
        \textcolor{gemaRed}{\textbf{CRASH se vazia!}}
      \end{alertblock}
    \column{0.31\textwidth}
      \begin{block}{top()}
        \small
        Retorna o topo sem remover.\\[4pt]
        \textbf{O(1)}\\[4pt]
        \textcolor{gemaRed}{\textbf{CRASH se vazia!}}
      \end{block}
  \end{columns}
  \vspace{0.2cm}
  \begin{columns}[t]
    \column{0.31\textwidth}
      \begin{block}{empty()}
        \small
        \texttt{true} se pilha vazia.\\[4pt]
        \textbf{Use SEMPRE antes} de \texttt{top()} e \texttt{pop()}.
      \end{block}
    \column{0.31\textwidth}
      \begin{block}{size()}
        \small
        Numero de elementos.\\[4pt]
        \textbf{O(1)}
      \end{block}
    \column{0.31\textwidth}
      \begin{tcolorbox}[caixaVerde,title={Dica de Prova}]
        \small
        Antes de qualquer acesso ao topo:\\[3pt]
        \textbf{sempre verifique}\\
        \texttt{!st.empty()}
      \end{tcolorbox}
  \end{columns}
\end{frame}

#include <bits/stdc++.h>
using namespace std;

int main() {
    stack<int> st; // pilha de inteiros

    st.push(10);   // insere 10 no topo
    st.push(20);   // insere 20 no topo
    st.push(30);   // insere 30 no topo

    // SEMPRE verifica antes de acessar!
    cout << st.top() << "\n"; // 30
    st.pop();                 // remove 30
    cout << st.top() << "\n"; // 20
    cout << st.size() << "\n";// 2

    while (!st.empty()) {
        cout << st.top() << " ";
        st.pop();
    }
    // saida final: 20 10
    return 0;
}

\begin{frame}[fragile,shrink=8]{Aplicacao Classica --- Parenteses Balanceados}
  \begin{columns}[c]
    \column{0.44\textwidth}
      \begin{tcolorbox}[caixaAzul,title={Problema}]
        \small Dada uma string com \texttt{( ) [ ] \{ \}},
        verificar se esta corretamente balanceada.
      \end{tcolorbox}
      \vspace{0.15cm}
      \textbf{Estrategia:}
      \begin{itemize}\small
        \item Abre $\to$ \texttt{push} na pilha
        \item Fecha $\to$ verifica topo e \texttt{pop}
        \item Ao final: pilha vazia = valido
      \end{itemize}
      \vspace{0.15cm}
      \begin{tikzpicture}[scale=0.90]

          at (0,0.8){\textbf{(\{[]\})} $\to$ VALIDO};
        \node[font=\ttfamily\small,gemaRed]
          at (0,0.2){\textbf{(\{[\}])} $\to$ INVALIDO};
        \node[font=\ttfamily\small,gemaRed]
          at (0,-0.4){\textbf{(()}\ \ \ \ $\to$ INVALIDO};
      \end{tikzpicture}
    \column{0.54\textwidth}
      \begin{tcolorbox}[caixaCodigo]
        \begin{lstlisting}[style=cppcompact]
bool valida(string s) {
    stack<char> st;
    for (char c : s) {
        // abre: empilha
        if (c=='('||c=='['||c=='{')
            st.push(c);
        else {
            if (st.empty()) return false;
            char t = st.top();
            st.pop();
            if (c==')' && t!='(') return false;
            if (c==']' && t!='[') return false;
            if (c=='}' && t!='{') return false;
        }
    }
    // sobrou na pilha = invalido
    return st.empty();
}
        \end{lstlisting}
      \end{tcolorbox}
  \end{columns}
\end{frame}

\begin{frame}{Motivacao --- Por que Stack Monotonica?}
  \begin{tcolorbox}[caixaAzul,title={Problema Central}]
    Dado um array, para cada posicao $i$ encontre o
    \textbf{proximo elemento maior} a direita.\\[4pt]
    Exemplo: \quad
    \texttt{[4,\ 5,\ 2,\ 10,\ 8]}
    $\;\longrightarrow\;$
    \texttt{[5,\ 10,\ 10,\ -1,\ -1]}
  \end{tcolorbox}
  \vspace{0.2cm}
  \begin{columns}[t]
    \column{0.46\textwidth}
      \begin{alertblock}{Forca Bruta --- O($n^2$)}
        \begin{itemize}\small
          \item Para cada \texttt{i}, varre todos \texttt{j > i}
          \item $n = 10^5 \Rightarrow 10^{10}$ operacoes
          \item $n = 5{\times}10^5 \Rightarrow$ \textbf{250 bilhoes!}
          \item \textbf{TLE garantido}
        \end{itemize}
      \end{alertblock}
    \column{0.50\textwidth}
      \begin{exampleblock}{Stack Monotonica --- O($n$)}
        \begin{itemize}\small
          \item Cada elemento entra na pilha \textbf{1 vez}
          \item Cada elemento sai da pilha \textbf{1 vez}
          \item Total: no maximo $2n$ operacoes
          \item \textbf{Argumento de amortizacao}
        \end{itemize}
      \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}[fragile,shrink=8]{Esqueleto Universal --- Decore Este Padrao}
  \begin{tcolorbox}[caixaAzul,title={Padrao que resolve 80\% dos problemas de stack monotonica}]
    \small
    O mesmo esqueleto resolve proximo maior, proximo menor, anterior maior,
    anterior menor --- basta trocar o operador e a direcao do loop.
  \end{tcolorbox}
  \vspace{0.08cm}
  \begin{tcolorbox}[caixaCodigo]
    \begin{lstlisting}[style=cppcompact]
stack<int> st;             // guarda INDICES (nunca valores!)
vector<int> resp(n, -1);   // resposta padrao = -1

for (int i = 0; i < n; i++) {
    // enquanto novo elemento "quebra" a monotonia da pilha:
    while (!st.empty() && arr[st.top()] < arr[i]) {
        // arr[i] e o "proximo maior" do indice no topo
        resp[st.top()] = arr[i];  // ou = i, depende do problema
        st.pop();
    }
    st.push(i);  // empilha o indice atual
}
// quem sobrou na pilha -> resp ja e -1
    \end{lstlisting}
  \end{tcolorbox}
  \vspace{0.05cm}
  \begin{tcolorbox}[caixaAmarelo,title={Por que guardar indices e nao valores?}]
    \small
    Com o \textbf{indice} voce acessa \texttt{arr[idx]}, \texttt{resp[idx]},
    a posicao original --- qualquer informacao.
    Com o \textbf{valor} voce perde a posicao e nao consegue atualizar \texttt{resp}.
  \end{tcolorbox}
\end{frame}

\begin{frame}[fragile,shrink=10]{Next Greater Element --- Codigo Completo}

        \begin{lstlisting}[style=cppcompact]
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    vector<int> a(n), resp(n, -1);
    for(int i = 0; i < n; i++) cin >> a[i];

    stack<int> st; // indices pendentes

    for(int i = 0; i < n; i++){
        // a[i] maior que o topo?
        // -> a[i] e o proximo maior do topo
        while(!st.empty() && a[i] > a[st.top()]){
            resp[st.top()] = a[i];
            st.pop();
        }
        st.push(i);
    }

    // quem sobrou: sem proximo maior
    for(int i = 0; i < n; i++)
        cout << resp[i] << " \n"[i==n-1];

    return 0;
}

\begin{frame}[fragile,shrink=12]{Maior Retangulo no Histograma --- Codigo}
  \begin{tcolorbox}[caixaCodigo]
    \begin{lstlisting}[style=cppcompact]
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n; cin >> n;
    vector<long long> h(n);
    for(int i = 0; i < n; i++) cin >> h[i];

    stack<int> st;       // stack CRESCENTE de indices
    long long maximo = 0;

    // sentinela 0 no final para esvaziar a pilha
    for(int i = 0; i <= n; i++){
        long long cur = (i == n) ? 0 : h[i];

        // cur menor que topo -> topo encontrou proximo menor
        while(!st.empty() && h[st.top()] > cur){
            long long altura = h[st.top()];
            st.pop();

            // anterior menor = novo topo (ou -1 se vazia)
            long long esq = st.empty() ? 0 : st.top() + 1;
            long long largura = i - esq;
            maximo = max(maximo, altura * largura);
        }

        st.push(i);
    }

    cout << maximo << "\n";
    return 0;
}
    \end{lstlisting}
  \end{tcolorbox}
  \vspace{0.05cm}
  \begin{tcolorbox}[caixaVerde,title={h = [2,1,5,6,2,3] --- Saida: 10}]
    \footnotesize
    A sentinela \texttt{cur=0} no final forca o esvaziamento da pilha,
    calculando a area de quem ainda estava pendente.
  \end{tcolorbox}
\end{frame}

#include <bits/stdc++.h>
using namespace std;

int main(){

\begin{frame}[shrink=4]{Como Identificar na Prova?}
  \begin{columns}[t]
    \column{0.47\textwidth}
      \begin{block}{Palavras-chave no enunciado}
        \begin{itemize}\small
          \item ``proximo maior / menor elemento''
          \item ``primeiro a esquerda / direita que\ldots''
          \item ``area maxima'', ``retangulo''
          \item ``visibilidade'' (predios, montanhas)
          \item ``dominancia entre elementos''
          \item ``temperatura mais alta antes de''
        \end{itemize}
      \end{block}
    \column{0.50\textwidth}
      \begin{exampleblock}{Sinais na estrutura do problema}
        \begin{itemize}\small
          \item $n \le 10^5$ com TL curto $\Rightarrow$ O($n^2$) nao passa
          \item ``Para cada posicao, encontre\ldots''
          \item Resposta de $i$ depende de min/max de um intervalo
          \item Versao facil ($n \le 1000$) existe
          \item Versao dificil pede O($n$)
        \end{itemize}
      \end{exampleblock}
  \end{columns}
  \vspace{0.15cm}
  \begin{tcolorbox}[caixaAmarelo,title={Armadilha Classica das Maratonas}]
    \small
    A versao facil aceita O($n^2$). A versao dificil tem $n \le 5{\times}10^5$
    e exige stack monotonica. \textbf{Aprenda O($n$) desde o inicio}.
  \end{tcolorbox}
\end{frame}

\begin{frame}[shrink=5]{Dicas de Prova}
  \begin{columns}[t]
    \column{0.47\textwidth}
      \begin{tcolorbox}[caixaVerde,title={Boas Praticas}]
        \begin{itemize}\small
          \item Guarde \textbf{indices} na pilha, nao valores
          \item Verifique \texttt{!st.empty()} antes de \texttt{top()} e \texttt{pop()}
          \item Defina o valor padrao da resposta antes do loop
          \item Teste: crescente, decrescente, constante, tamanho 1
          \item Use \texttt{long long} quando valores forem grandes
          \item Sentinela esvazia a pilha no final
        \end{itemize}
      \end{tcolorbox}
    \column{0.50\textwidth}
      \begin{tcolorbox}[caixaVermelho,title={Erros Frequentes}]
        \begin{itemize}\small
          \item \texttt{top()} ou \texttt{pop()} em pilha vazia $\to$ \textbf{crash}
          \item Usar \texttt{<} quando devia ser \texttt{<=}
          \item Esquecer indices que sobraram na pilha
          \item Confundir ``proximo'' com ``anterior''
          \item Saida em base 0 quando pede base 1
        \end{itemize}
      \end{tcolorbox}
  \end{columns}
  \vspace{0.12cm}
  \begin{tcolorbox}[caixaAzul,title={Template Mental --- decore este padrao}]
    \centering\ttfamily\small
    while (!st.empty() \&\& CONDICAO) \{\ REGISTRA;\ st.pop();\ \}\ st.push(i);
  \end{tcolorbox}
\end{frame}

  \begin{columns}[t]

      \begin{block}{Stack Classica}
        \begin{itemize}\small
          \item LIFO --- Last In, First Out
          \item Operacoes O(1): \texttt{push}, \texttt{pop}, \texttt{top}, \texttt{empty}
          \item Usos: parenteses, undo/redo, expressoes
          \item \texttt{std::stack<T>} na STL C++

          \item CF 1313C2 --- Skyscrapers Hard
          \item CF 547B --- Nearest Smaller
          \item CF 281D --- Maximum Xor Secondary

    \column{0.50\textwidth}
      \begin{block}{Stack Monotonica}
        \begin{itemize}\small
          \item Mantem ordem crescente ou decrescente
          \item Max. 1 push + 1 pop por elemento $\Rightarrow$ \textbf{O($n$)}
          \item Decrescente $\to$ proximo \textbf{maior}
          \item Crescente $\to$ proximo \textbf{menor}
          \item Loop invertido $\to$ anterior maior/menor

      \begin{tcolorbox}[caixaVermelho,title={Regra de Ouro}]

        \textbf{Ao fazer pop}, o elemento atual e a resposta para o indice que saiu.\\[3pt]
        Sobrou na pilha $\Rightarrow$ resposta e \textbf{-1}.
